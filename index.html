                    } else {

                        currentDayData.value.items.push({

                            id: Date.now().toString(),

                            ...formSchedule.value

                        });

                    }

                    modals.value.schedule = false;

                };


                const confirmDeleteSchedule = () => {

                    if (confirm('確定要刪除這個行程地標嗎？此操作無法還原。')) {

                        currentDayData.value.items.splice(editingIndex, 1);

                        modals.value.schedule = false;

                    }

                };


                const handleImageUpload = (e) => {

                    const file = e.target.files[0];

                    if (file) {

                        formShopping.value.image = URL.createObjectURL(file);

                    }

                };


                const saveShopping = () => {

                    if (!formShopping.value.name.trim()) return alert('請輸入想買的物品名稱');

                    shoppingList.value.push({ ...formShopping.value });

                    modals.value.shopping = false;

                };


                const deleteShoppingItem = (idx) => {

                    if(confirm('確認從清單移除？')) shoppingList.value.splice(idx, 1);

                };


                const saveExpense = () => {

                    if (!formExpense.value.name.trim() || !formExpense.value.amount) return alert('名稱與金額為必填項目');

                    expensesList.value.push({ ...formExpense.value });

                    modals.value.expenses = false;

                };


                const deleteExpense = (idx) => {

                    if(confirm('要刪除這筆花費紀錄嗎？')) expensesList.value.splice(idx, 1);

                };


                // 優化過後的 Google Map 喚醒導航函數

                const openGoogleMap = (keyword) => {

                    const cleanKeyword = encodeURIComponent(keyword);

                    // 同時支援 iOS / Android 手機原生 App 與網頁跳轉

                    const url = `https://www.google.com/maps/search/?api=1&query=${cleanKeyword}`;

                    window.open(url, '_blank');

                };

                        '航班': 'bg-emerald-50 text-emerald-700 border border-emerald-200/60',

                        '其他': 'bg-slate-50 text-slate-600 border border-slate-200/60'

                    };

                    return maps[cate] || maps['其他'];

                const getTransportIcon = (type) => {

                    if (type === 'walk') return 'fa-solid fa-person-walking text-emerald-500';

                    if (type === 'transit') return 'fa-solid fa-bus text-orange-500';

                    return 'fa-solid fa-car-side text-[#2B5C8F]';

                };


                const getExpenseIcon = (cate) => {

                    const maps = { '交通': 'fa-solid fa-car', '飲食': 'fa-solid fa-utensils', '住宿': 'fa-solid fa-bed', '門票': 'fa-solid fa-ticket', '購物': 'fa-solid fa-bag-shopping', '其他': 'fa-solid fa-ellipsis' };

                    return maps[cate] || 'fa-solid fa-receipt';

                };


                const getExpenseIconClass = (cate) => {

                    const maps = {

                        '交通': 'bg-blue-50 text-blue-600',

                        '飲食': 'bg-rose-50 text-rose-600',

                        '住宿': 'bg-purple-50 text-purple-600',

                        '門票': 'bg-amber-50 text-amber-600',

                        '購物': 'bg-orange-50 text-orange-600',

                        '其他': 'bg-slate-100 text-slate-600'

                    };

                    return maps[cate] || 'bg-slate-100';

                };


                const initSortable = () => {

                    const el = document.getElementById('drag-container');

                    if (!el) return;

                    

                    Sortable.create(el, {

                        handle: '.handle',

                        animation: 200,

                        ghostClass: 'bg-blue-50',

                        onEnd: (evt) => {

                            const items = [...currentDayData.value.items];

                            const [movedItem] = items.splice(evt.oldIndex, 1);

                            items.splice(evt.newIndex, 0, movedItem);

                            currentDayData.value.items = items;

                        }

                    });

                };


                onMounted(() => {

                    initSortable();

                    watch(selectedDayIndex, () => {

                        nextTick(() => initSortable());

                    });

                    watch(currentTab, (newTab) => {

                        if (newTab === 'schedule') {

                            nextTick(() => initSortable());

                        }

                    });

                });


                return {

                    currentTab, selectedDayIndex, scheduleData, currentDayData,

                    shoppingList, expensesList, totalExpenses, modals, isEditing,

                    formSchedule, formShopping, formExpense, exchange, weatherInfo,

                    calcCurrency, openAddModal, triggerEditItem, saveSchedule, confirmDeleteSchedule,

                    handleImageUpload, saveShopping, deleteShoppingItem, saveExpense, deleteExpense,

                    openGoogleMap, getCateBadgeClass, getTransportIcon, getExpenseIcon, getExpenseIconClass

                };

            }

        }).mount('#app');
